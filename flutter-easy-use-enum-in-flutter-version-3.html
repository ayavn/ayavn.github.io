enum HogeType {
  hoge('汇', 'hoge'),
  fuga('点', 'fuga'),
  piyo('淬', 'piyo'),
  foo('点', 'foo'),
  bar('般', 'bar'),
  none('', '');

  const HogeType(this.name, this.type);
  final String name;
  final String type;

  static HogeType fromType(String from) {
    return HogeType.values.firstWhere(
      (value) {
        return value.type == from;
      },
      orElse: () {
        return HogeType.none;
      },
    );
  }
}

void hg() {
  print(HogeType.fromType('hoge'));
  //HogeType.hoge
  print(HogeType.fromType('aa'));
  //HogeType.none
}
